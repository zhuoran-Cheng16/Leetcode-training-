## Code

```python
class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        res=[]
        def dfs(p, num, left, right):
            if len(p)== n*2:# 长度为n*2时停止迭代
                res.append(p)
                return 
            if left<num: # 左括号数量不多于n
                dfs(p+'(',num, left+1, right)
            if right<left:
                dfs(p+')',num, left,right+1)
        
        dfs('',n,0,0)

        return res
```
